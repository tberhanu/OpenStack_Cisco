FROM containers.cisco.com/aws_managed/python-3.6:latest 
SHELL ["/bin/bash", "-c"]
USER root
MAINTAINER phuocng

# Set the localtime as per PST
# RUN ln -fs /usr/share/zoneinfo/America/Los_Angeles /etc/localtime

# Configure username and useremail required for Git
RUN git config --global user.name csbauditor.gen
RUN git config --global user.email csbauditor.gen@cisco.com

# Limiting the permissions on home dir
RUN chmod -R 775 $HOME

CMD python $HOME/p3_cae_list_projects_and_update_dynamoDB.py | /usr/bin/tee $HOME/logs/teamdb_update_log_$(date +"%m%d%y_%H%M%S")
