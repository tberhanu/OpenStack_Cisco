FROM containers.cisco.com/aws_managed/python-3.6:latest 
SHELL ["/bin/bash", "-c"]
USER root
MAINTAINER Phuoc

# Configure username and useremail required for Git
RUN git config --global user.name csbauditor.gen
RUN git config --global user.email csbauditor.gen@cisco.com

# Limiting the permissions on home dir
RUN chmod -R 775 /opt/app-root/src/

# Installing the packages required to support overall execution
ADD requirements.txt /opt/app-root/src
RUN chmod 755 /opt/app-root/src/requirements.txt
RUN pip install --upgrade pip && pip install -r requirements.txt

# Create dir for storing the logs
RUN mkdir /opt/app-root/src/logs
RUN chmod -R 775 /opt/app-root/src/logs

# Adding required dependency files and limiting the permissions
ADD audit_projects.py /opt/app-root/src
RUN chmod 755 /opt/app-root/src/audit_projects.py
ADD csb_credentials.py.enc /opt/app-root/src
RUN chmod 755 /opt/app-root/src/csb_credentials.py.enc
ADD data_cae.xml /opt/app-root/src
RUN chmod 755 /opt/app-root/src/data_cae.xml
ADD data_p3.xml /opt/app-root/src
RUN chmod 755  /opt/app-root/src/data_p3.xml
ADD env_variables.py /opt/app-root/src
RUN chmod 755  /opt/app-root/src/env_variables.py
ADD generate_audit_report.py /opt/app-root/src
RUN chmod 755 /opt/app-root/src/generate_audit_report.py
ADD kube_config_alln.enc /opt/app-root/src
RUN chmod 755 /opt/app-root/src/kube_config_alln.enc
ADD kube_config_rcdn.enc /opt/app-root/src
RUN chmod 755  /opt/app-root/src/kube_config_rcdn.enc
ADD kube_config_rtp.enc /opt/app-root/src
RUN chmod 755 /opt/app-root/src/kube_config_rtp.enc
ADD p3_cae_list_projects_and_update_dynamoDB.py /opt/app-root/src
RUN chmod 755  /opt/app-root/src/p3_cae_list_projects_and_update_dynamoDB.py
ADD teams_db_update_util.py /opt/app-root/src/
RUN chmod 755 /opt/app-root/src/teams_db_update_util.py
CMD python /opt/app-root/src/p3_cae_list_projects_and_update_dynamoDB.py > /opt/app-root/src/logs/teamdb_update_log_$(date +"%m%d%y_%H%M%S")
